```python
import subprocess                          # patch: services.ansible.subprocess.Popen
from config import (ANSIBLE_DIR, ANSIBLE_PLAYS_DIR, ANSIBLE_ROLES_DIR,
                    ANSIBLE_INV_DIR, ANSIBLE_HVARS_DIR, ANSIBLE_GVARS_DIR)
from models import ansible as ansible_repo

def _generate_inventory() -> str:
    """Queries ansible_repo.get_hosts_active_with_groups() -> (hosts, groups, hg).
    Builds YAML inventory dict. Writes hosts.yml to ANSIBLE_INV_DIR.
    Writes group_vars and host_vars files to ANSIBLE_GVARS_DIR and ANSIBLE_HVARS_DIR.
    Returns inventory YAML string."""

def _do_ansible_run(run_id, playbook_path, extra_args: list):
    """Background thread function. Calls:
    1. _set_run_status(run_id, 'running')
    2. ansible_repo.set_run_started(run_id)
    3. _generate_inventory()
    4. subprocess.Popen(['ansible-playbook', '-i', inv_path, playbook_path] + extra_args)
    5. Reads stdout line by line, appends to run log
    6. On exit_code==0: _set_run_status(run_id, 'success', 0)
       On exit_code!=0: _set_run_status(run_id, 'failed', exit_code)
    7. On FileNotFoundError: logs error, _set_run_status(run_id, 'failed', -1)"""
```

```python
def test_do_ansible_run_success(app_with_db):
    # Seed a run record
    run_id = ansible_repo.create_run(
        playbook_id=1, playbook_name='test.yml', inventory='',
        extra_vars='', limit_hosts='', tags_run='', triggered_by='test'
    )

    mock_proc = MagicMock()
    mock_proc.pid = 12345
    mock_proc.stdout = iter(['PLAY [all] ***\n', 'ok: [host1]\n'])
    mock_proc.wait.return_value = 0
    mock_proc.returncode = 0

    with patch.object(ansible_service, '_generate_inventory', return_value='---\nall: {}'), \
         patch('services.ansible.subprocess.Popen', return_value=mock_proc):
        ansible_service._do_ansible_run(run_id, '/path/test.yml', [])

    row = ansible_repo.get_run_by_id(run_id)
    assert row['status'] == 'success'
    assert row['exit_code'] == 0
```

For failure test: mock_proc.returncode = 2, mock_proc.wait.return_value = 2. Assert status='failed', exit_code=2.
