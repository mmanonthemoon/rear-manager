```python
# Imports:
from services import ssh as ssh_service    # patch target: services.rear.ssh_service.*
from services import jobs as job_service    # patch target: services.rear.job_service.*
from models import jobs as job_repo         # patch target: services.rear.job_repo.*
from models import settings as settings_repo
from models import servers as server_repo

def generate_rear_config(server, cfg, extra_server_exclude='') -> str:
    """Pure-ish function. Calls settings_repo.get_nfs_target(server['hostname'], _get_local_ip, _safe_dirname, BACKUP_ROOT).
    server: dict with 'hostname', 'exclude_dirs' keys
    cfg: dict with 'autoresize', 'migration_mode', 'rear_output', 'rear_backup', 'global_exclude_dirs' keys
    Returns: string content of /etc/rear/local.conf"""

def _run_install_rear(job_id, server_dict):
    """Background thread function. Calls:
    1. job_service._set_job_status(job_id, 'running')
    2. job_repo.set_started(job_id)
    3. ssh_service.ssh_get_os_info(server_dict) -> detects OS
    4. ssh_service.ssh_exec_stream(server_dict, apt_cmd/dnf_cmd, log) -> install
    5. ssh_service.ssh_exec_stream(server_dict, 'rear --version', log) -> verify
    6. job_service._set_job_status(job_id, 'success')"""

def _run_configure_rear(job_id, server_dict, rear_config_content):
    """Background thread function. Calls:
    1. job_service._set_job_status(job_id, 'running')
    2. job_repo.set_started(job_id)
    3. ssh_service.ssh_exec_stream(server_dict, 'mkdir -p /etc/rear', log)
    4. ssh_service.ssh_exec_stream(server_dict, 'test -f ... cp ...', log) -> backup existing
    5. ssh_service.ssh_upload_file(server_dict, rear_config_content, '/etc/rear/local.conf')
    6. ssh_service.ssh_exec_stream(server_dict, 'rear dump ...', log) -> verify
    7. server_repo.update_rear_configured(server_dict['id'])
    8. job_service._set_job_status(job_id, 'success')"""
```
